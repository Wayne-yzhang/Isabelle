  "(\<ge>)\<^sup>*\<^sup>* = (\<ge>)"
  by (intro ext) (auto elim: rtranclp_induct)

lemma (in order) tranclp_greater [simp]:
  "(>)\<^sup>+\<^sup>+ = (>)"
  by (intro ext) (auto elim: tranclp_induct)
