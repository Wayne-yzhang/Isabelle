theory State
imports
  "../Common/Heap"

type_synonym

definition hp :: "'heap \<times> 'x \<Rightarrow> 'heap" where "hp \<equiv> fst"

definition lcl :: "'heap \<times> 'x \<Rightarrow> 'x" where "lcl \<equiv> snd"

lemma hp_conv [simp]: "hp (h, l) = h"
by(simp add: hp_def)

lemma lcl_conv [simp]: "lcl (h, l) = l"
by(simp add: lcl_def)

end
